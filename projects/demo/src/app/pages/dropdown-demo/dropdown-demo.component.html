Component

import { Component, signal } from '@angular/core';
import { NgxTableLayoutPickerDropdownComponent, TableSelection } from 'ngx-table-layout-picker';

@Component({
  selector: 'app-example',
  imports: [NgxTableLayoutPickerDropdownComponent],
  template: `
    <ngx-table-layout-picker-dropdown
      buttonLabel="Insert Table"
      buttonIcon="table_chart"
      buttonAppearance="raised"
      [selectorConfig]="{
        rows: 10,
        cols: 10,
        theme: 'auto',
        showFooter: true
      }"
      (tableSelected)="onSelection($event)">
    </ngx-table-layout-picker-dropdown>
  `
})
export class ExampleComponent {
  protected readonly selection = signal<TableSelection | null>(null);

  protected onSelection(selection: TableSelection): void {
    this.selection.set(selection);
    console.log('Selected:', selection);
  }
}
 Configuration Options

Dropdown Component

  • buttonLabel (string): Text displayed on the button (default: 'Select Table')
  • buttonIcon (string): Material icon name for the button (optional)
  • buttonAppearance ('basic' | 'raised' | 'stroked' | 'flat'): Button style (default: 'basic')
  • buttonColor ('primary' | 'accent' | 'warn'): Button color (optional)
  • selectorConfig (TableLayoutConfig): Configuration for the picker component
  • tableSelected (event): Emits TableSelection when user selects dimensions

Selector Config

  • rows (number): Number of rows in the grid (default: 10)
  • cols (number): Number of columns in the grid (default: 10)
  • theme ('light' | 'dark' | 'auto'): Color theme (default: 'auto')
  • showFooter (boolean): Display selection text (default: true)
  • responsive (boolean): Enable adaptive grid sizing (default: false)
